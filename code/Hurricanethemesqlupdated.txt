SELECT (a.DATEADDED), GLOBALEVENTID, ActionGeo_CountryCode, ActionGeo_Type, ActionGeo_Lat, ActionGeo_Long, SQLDATE, SOURCEURL
FROM [gdelt-bq:gdeltv2.events] a
JOIN EACH (
 SELECT DATEADDED
 FROM [gdelt-bq:gdeltv2.events]
 WHERE DATEADDED>=20190901000000 and DATEADDED<=20190906234500 
 GROUP BY 1
) b
ON a.DATEADDED=b.DATEADDED
WHERE GLOBALEVENTID IN (
 SELECT GLOBALEVENTID 
 FROM [gdelt-bq:gdeltv2.eventmentions] 
 WHERE MentionIdentifier IN (
 SELECT DocumentIdentifier 
 FROM [gdelt-bq:gdeltv2.gkg] 
 WHERE V2Themes CONTAINS 'HURRICANE' AND DATE>=20190901000000 and DATE<=20190906234500
 )
 GROUP BY GLOBALEVENTID
) 
AND a.DATEADDED>20190831234500 
GROUP BY (a.DATEADDED), GLOBALEVENTID, ActionGeo_Type, ActionGeo_CountryCode, ActionGeo_Lat, ActionGeo_Long, SQLDATE, SOURCEURL
